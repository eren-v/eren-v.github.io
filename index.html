PDF to JPG Conversion with C#

Convert Pdf file to Jpg using GhostScript | March 24, 2017

This function gets pdf file from directory, call ghostscript and convert by parameters given "String ars =..."

public string PdfToJpg(string inputPDFFile, string outputImagesPath)
{
var projectpath = new DirectoryInfo(string.Format("{0}bin\\", Server.MapPath(@"\")));
string ghostScriptPath = System.IO.Path.Combine(projectpath.ToString(), "gs\\gs9.19\\bin\\gswin32.exe");

string outputname = inputPDFFile.Replace(".pdf", String.Empty);
//string ghostScriptPath = @"..\bin\gs\gs9.19\bin\gswin32.exe";
String ars = "-dNOPAUSE -sDEVICE=jpeg -dJPEGQ=100 -r300 -o" + outputImagesPath + outputname + ".jpg -sPAPERSIZE=a4 " + outputImagesPath + inputPDFFile;
Process proc = new Process();
proc.StartInfo.FileName = ghostScriptPath;
proc.StartInfo.Arguments = ars;
proc.StartInfo.CreateNoWindow = true;
proc.StartInfo.WindowStyle = ProcessWindowStyle.Hidden;
proc.Start();
proc.WaitForExit();

string newPath = outputname + ".jpg";
return newPath;
}

# Note that unrelated, unchecked-in items from other solutions may appear in the list of items to check in, but only those files from the current solution will have check marks. There is no need to change anything.
